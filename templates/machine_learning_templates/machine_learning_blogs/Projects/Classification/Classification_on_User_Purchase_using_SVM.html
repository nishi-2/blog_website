Using GridSearchCV

In[27]:
hyperparameters = [
    {'kernel' : ['linear'], 'C' : [1,10]},
    {'kernel' : ['poly'], 'C':[0.1,1,10]},
    {'kernel' : ['sigmoid'], 'C':[0.1,1,10]},
    {'kernel' : ['rbf'], 'gamma':[1e-3, 1e-4], 'C':[1,10]}
] 


# Define custom scorers with zero_division=0 for GridSearchCV
scorers = {
    'precision': make_scorer(precision_score, zero_division=0, average='binary'),
    'recall': make_scorer(recall_score, zero_division=0, average='binary')
}

scores = ['precision', 'recall']

for score in scores:
    print('Tuning hyperparameter for', score)
    print()

    clf = GridSearchCV(SVC(), hyperparameters, scoring=scorers[score])
    clf.fit(x_train_rescaled, y_train)

    print('Best parameters found:\n', clf.best_params_)
    print()
    print('Grid score on development set:\n')
    means = clf.cv_results_['mean_test_score']
    stds = clf.cv_results_['std_test_score']
    for mean, std, params in zip(means, stds, clf.cv_results_['params']):
        print(mean, std * 2, params)
    print()

    print('Detailed Classification Report:\n')
    y_true, y_pred = y_test, clf.predict(x_test_rescaled)
    print(classification_report(y_true, y_pred, zero_division=0))
    print('\n')
